// server.js (ESM)
import dotenv from 'dotenv';
dotenv.config();

import express from 'express';
import cors from 'cors';
import morgan from 'morgan';
import cookieParser from 'cookie-parser';
import mongoose from 'mongoose';

import authRoutes from './routes/authRoutes.js';
// (optional) import other routers when ready
// import seedRoutes from './routes/seed.js';
// import courseRoutes from './routes/courseRoutes.js';
// import quizRoutes from './routes/quizRoutes.js';

const app = express();

// ---------- core middleware (MUST come before routes) ----------
app.use(cors({ origin: true, credentials: true }));
app.use(morgan('dev'));
app.use(express.json({ limit: '1mb' }));
app.use(express.urlencoded({ extended: true }));
app.use(cookieParser());

// ---------- health ----------
app.get('/healthz', (_req, res) => res.json({ ok: true }));

// ---------- DEBUG: echo (used by your PowerShell smoke) ----------
app.post('/api/_debug/echo', (req, res) => {
  console.log('DEBUG echo content-type:', req.headers['content-type']);
  console.log('DEBUG echo body:', req.body);
  res.json({ ok: true, body: req.body ?? null });
});

const { attachUser } = require("./middleware/auth");
app.use(attachUser);

// ---------- mount routers ----------
app.use('/api/auth', authRoutes);
// app.use('/api/seed', seedRoutes);
// app.use('/api/courses', courseRoutes);
// app.use('/api/quizzes', quizRoutes);

// ---------- 404 ----------
app.use((req, res, next) => {
  if (res.headersSent) return next();
  res.status(404).json({ success: false, message: `Not found: ${req.method} ${req.originalUrl}` });
});

// ---------- error handler ----------
app.use((err, _req, res, _next) => {
  console.error('âŒ Unhandled error:', err);
  const status = err.status || 500;
  res.status(status).json({ success: false, message: err.message || 'Internal Server Error' });
});

// ---------- start ----------
const PORT = Number(process.env.PORT || 10003);
const MONGO_URI = process.env.MONGO_URI || 'mongodb://127.0.0.1:27017/microcourse';

(async () => {
  try {
    await mongoose.connect(MONGO_URI, { dbName: 'microcourse' });
    console.log('âœ… Mongo connected');
    app.listen(PORT, () => console.log(`ğŸš€ Server running on port ${PORT}`));
  } catch (e) {
    console.error('âŒ Startup failure:', e);
    process.exit(1);
  }
})();
